1"""Session-id detection for Claude Code processes. 

2 

3Claude Code's Bash tool runs without CLAUDE_SESSION_ID exported, so we 

4infer the session by walking up the process tree looking for marker 

5files written by the cost-discipline hook or by our own register 

6command.""" 

7from __future__ import annotations 

8 

9import os 

10import subprocess 

11from pathlib import Path 

12 

13from . import paths

15_ANCESTOR_HOPS = 10

18def _walk_ancestors(): 

19 pid = os.getpid() 

20 for _ in range(_ANCESTOR_HOPS): 

21 try: 

22 ppid = int(subprocess.check_output( 

23 ["ps", "-o", "ppid=", "-p", str(pid)], 

24 stderr=subprocess.DEVNULL, 

25 ).decode().strip()) 

26 except Exception: 

27 return 

28 if ppid <= 1: 

29 return 

30 yield ppid 

31 pid = ppid

34def _process_is_claude(pid: int) -> bool: 

35 """Return True if pid is a currently-running 'claude' process. 

36 Used to filter stale markers — a marker is only trustworthy if 

37 its owning PID is still alive AND still the claude that wrote it. 

38 PIDs get recycled, so 'alive' alone is not enough.""" 

39 try: 

40 comm = subprocess.check_output( 

41 ["ps", "-o", "comm=", "-p", str(pid)], 

42 stderr=subprocess.DEVNULL, 

43 ).decode().strip().lower() 

44 except Exception: 

45 return False 

46 # `ps -o comm=` sometimes reports the full resolved binary path rather 

47 # than just the short process name — and where "claude" lands in that 

48 # path varies: 

49 # - the real Claude Code install resolves to 

50 # .../local/share/claude/versions/<version> — "claude" is a middle 

51 # PATH SEGMENT, while the basename is just the version number 

52 # (e.g. "2.1.197"). Basename-only matching false-negatives here. 

53 # - a process invoked via `uv run` from a checkout directory named 

54 # "downbeat" resolves to .../downbeat/.venv/bin/python3 — 

55 # "claude" is a substring of an unrelated directory NAME 

56 # ("downbeat"), not an exact segment. Whole-string substring 

57 # matching false-positives here. 

58 # Matching an EXACT path segment (split on "/") handles both: it accepts 

59 # "claude" as a full segment anywhere in the path, but rejects a 

60 # look-alike segment like "downbeat". 

61 return "claude" in comm.split("/")

64def detect_session_id() -> str | None: 

65 """Return the current Claude Code session_id, or None if not detectable. 

66 

67 Only trusts markers whose owning PID is BOTH alive AND a claude 

68 process. Stale markers (PID dead, or PID recycled to a non-claude 

69 process) are skipped — they would otherwise misattribute messages 

70 to peers that no longer correspond to the calling session.""" 

71 for pid in _walk_ancestors(): 

72 if not _process_is_claude(pid): 

73 continue 

74 # 1) cost-discipline hook marker 

75 cd_marker = Path(f"/tmp/cc-session-by-pid-{pid}.txt") 

76 if cd_marker.exists(): 

77 return cd_marker.read_text().strip() 

78 # 2) our own register backstop 

79 # Per-pid relay markers are written by `register`; the python CLI 

80 # process is short-lived so its PID rarely matches an ancestor. 

81 # Kept for backwards compat with older installs. 

82 relay_marker = paths.RELAY_DIR / f".sid-{pid}" 

83 if relay_marker.exists(): 

84 return relay_marker.read_text().strip() 

85 return None

88def gc_stale_markers() -> dict[str, int]: 

89 """Remove marker files whose owning PID is dead OR no longer a claude 

90 process. Returns counts of removed files per location.""" 

91 removed = {"tmp": 0, "relay": 0} 

92 # Sweep /tmp/cc-session-by-pid-*.txt 

93 for p in Path("/tmp").glob("cc-session-by-pid-*.txt"): 

94 try: 

95 pid = int(p.stem.split("-")[-1]) 

96 except ValueError: 

97 continue 

98 if not _process_is_claude(pid): 

99 try: 

100 p.unlink() 

101 removed["tmp"] += 1 

102 except OSError: 

103 pass 

104 # Sweep ~/.claude/relay/.sid-* 

105 if paths.RELAY_DIR.exists(): 

106 for p in paths.RELAY_DIR.glob(".sid-*"): 

107 try: 

108 pid = int(p.name.removeprefix(".sid-")) 

109 except ValueError: 

110 continue 

111 if not _process_is_claude(pid): 

112 try: 

113 p.unlink() 

114 removed["relay"] += 1 

115 except OSError: 

116 pass 

117 return removed

126def process_start_time(pid: int) -> str | None: 

127 """Return ISO-8601 normalized process start time, or None. 

128 macOS: `ps -o lstart=` returns 'Mon May 27 09:11:11 2026'. 

129 Linux: `ps -o lstart=` returns the same format on most distros. 

130 We parse and re-emit as ISO-8601 with second precision.""" 

131 try: 

132 raw = subprocess.check_output( 

133 ["ps", "-o", "lstart=", "-p", str(pid)], 

134 stderr=subprocess.DEVNULL, 

135 ).decode().strip() 

136 except Exception: 

137 return None 

138 if not raw: 

139 return None 

140 try: 

141 from datetime import datetime 

142 dt = datetime.strptime(raw, "%a %b %d %H:%M:%S %Y") 

143 return dt.isoformat(timespec="seconds") 

144 except ValueError: 

145 return None
